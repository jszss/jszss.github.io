    //获取当前时间-年月日星期时分秒
function getDate() {
    var today = new Date();
    var time = twoDigits(today.getHours()) + ":" + twoDigits(today.getMinutes());
    $(".date_time").html(time);
}

function twoDigits(val) {
    if (val < 10) return "0" + val;
    return val;
}
getDate();

//每秒获取执行一次，更新时间
$(function() {
    setInterval(getDate, 1000);
});
