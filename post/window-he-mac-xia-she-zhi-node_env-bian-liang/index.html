    • window和mac下设置NODE_ENV变量

    mac export设置变量

    "scripts": {
   "start": "export NODE_ENV=development && nodemon -w src --exec \"babel-node src\"",
   "build": "babel src --out-dir dist",
   "run-build": "node dist",
 }

    window set设置变量

    "scripts": {
   "start": "set NODE_ENV=development && nodemon -w src --exec \"babel-node src\"",
   "build": "babel src --out-dir dist",
   "run-build": "node dist",
 }
